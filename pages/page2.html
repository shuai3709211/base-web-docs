import { type TableColumnProps } from 'base-web/packages/components';
  import { computed, onMounted, ref } from 'vue';
  import {
    addApi,
    addEnvironmentItemApi,
    checkEnvironmentItemCodeRepeatApi,
    delEnvironmentApi,
    delEnvironmentItemApi,
    getEnvironmentFormApi,
    getEnvironmentItemListApi,
    getEnvironmentListApi,
    getEnvironmentRepeatApi,
    updateEnvironmentApi,
    updateEnvironmentItemApi
  } from '@/api/xx/xxxApi';
  import { confirm, msgSuccess, showFormDrawer, type MapItem } from 'base-web/packages/utils';


  defineOptions({
    name: 'EnvManage'
  });


  const search = ref({});//原始搜索表单
  const searchColumns = ref([{
    el: 'input',
    prop: 'name|code',
    placeholder: '请输入分类名称/编码',
    isKeywords: true,//是否是关键字搜索
    span: 18
  }]);


  /**转换后的搜索表单*/
  const transformSearch = ref({});

  /**列表加载动画*/
  const loading = ref(true); //
  /**列表双向绑定值*/
  const tables = ref < MapItem[] > (); //
  /**总条数*/
  const total = ref(0);
  /**当前第几条*/
  const pageNum = ref(1); //
  /**每页多少条*/
  const pageSize = ref(10); //

  /**抽屉表单描述*/
  const drawerColumns = computed(() => {
    return [
      {
        el: 'input',
        prop: 'name',
        label: '分类名称',
        rules: { required: true, message: '分类名称不能为空', trigger: 'submit' }
      },
      {
        el: 'input',
        prop: 'code',
        label: '分类编码',
        rules: {
          required: true, validator: async (rule, value, callback) => {
            if (!value) {
              callback(new Error('分类编码不能为空'));
            } else {
              const res = await getEnvironmentRepeatApi(value, selectRow.value.id);
              if (res.data) {
                //重复
                callback(new Error('分类编码重复'));
              }
            }
          }, trigger: 'submit'
        }
      }

    ];
  });

  /**选中行*/
  const selectRow = ref < MapItem > ({});
  const currRow = ref < MapItem > ({});


  const tableColumns: TableColumnProps[] = [
    { prop: 'name', label: '分类名称' },
    { prop: 'code', label: '分类编码' },
    { prop: 'options', label: '操作', width: 140 }
  ];


  const createTables = () => {
    tables.value = [];
    loading.value = true;
    getEnvironmentListApi({
      ...transformSearch.value,
      pageNum: pageNum.value,
      pageSize: pageSize.value
    }).then(({ data }) => {
      tables.value = data?.records || data || [];
      total.value = Number(data?.total || 0);
    }).finally(() => {
      loading.value = false;

    });
  };


  const columnHandler = async (event: string, row
    ? : any;
  ) =>
  {
    selectRow.value = { ...row };
    if ((event == 'edit' || event == 'view')) {
      const res = await getEnvironmentFormApi(row.id);
      row = res?.data || row;
    }
    if (event == 'add' || event == 'edit') {

      const info = row ? { ...row } : {};
      const { close } = showFormDrawer({
          title: event == 'add' ? '新增分类' : '修改分类',
          columns: drawerColumns.value,
          params: info,
          submit: (value) => {
            if (event == 'add') {
              addEnvironmentApi(value).then(() => {
                msgSuccess('新增成功');
                close();
                createTables();
              });
            } else {
              updateEnvironmentApi(value.id, value).then(() => {
                msgSuccess('修改成功');
                close();
                createTables();
              });
            }
          }
        }
      );
    } else if (event == 'delete') {
      const is = await confirm('确认删除已选中的数据项?', '警告');
      if (is == 'ok') {
        delEnvironmentApi(row.id).then(() => {
          msgSuccess('删除成功');
          createTables();
        });
      }

    }
  }
  ;


  const cellClick = ({ row }) => {
    currRow.value = row || {};
    createTables2();
  };


  const tableColumns2: TableColumnProps[] = [
    { prop: 'name', label: '参数名称' },
    { prop: 'code', label: '参数编码' },
    { prop: 'value', label: '参数值' },
    { prop: 'remark', label: '备注' },
    { prop: 'seq', label: '序号', width: 60 },
    { prop: 'options', label: '操作', width: 140 }
  ];
  const tables2 = ref([]);
  const loading2 = ref(false);
  const selectRow2 = ref < MapItem > ({});
  /**抽屉表单描述*/
  const drawerColumns2 = computed(() => {
    return [
      {
        el: 'input',
        prop: 'name',
        label: '参数名称',
        rules: { required: true, message: '请输入参数名称', trigger: 'submit' }
      },
      {
        el: 'input',
        prop: 'code',
        label: '参数编码',
        rules: {
          required: true, validator: async (_, value, callback) => {
            if (!value) {
              callback(new Error('请输入参数编码'));
            } else {
              const res = await checkEnvironmentItemCodeRepeatApi(value, currRow.value.id, selectRow2.value?.id);
              if (res.data) {
                callback(new Error('参数编码重复'));
              }
              callback();
            }
          }, trigger: 'submit'
        }
      },
      {
        el: 'input',
        prop: 'value',
        type: 'textarea',
        maxlength: 50,
        label: '参数值',
        rules: { required: true, message: '请输入参数值', trigger: 'submit' }
      }, {
        el: 'input',
        prop: 'remark',
        type: 'textarea',
        maxlength: 250,
        label: '备注'
      },
      {
        el: 'inputNumber',
        prop: 'seq',
        max: 99999,
        min: 0,
        label: '显示排序'
      }

    ];
  });

  const createTables2 = () => {
    tables2.value = [];
    loading2.value = true;
    getEnvironmentItemListApi(currRow.value.id).then(({ data }) => {
      tables2.value = data?.records || data || [];
    }).finally(() => {
      loading2.value = false;

    });
  };


  const columnHandler2 = async (event: string, row
    ? : any;
  ) =>
  {
    selectRow2.value = row || {};

    if (event == 'add' || event == 'edit') {
      const info = row ? { ...row } : { seq: 1 };
      const { close } = showFormDrawer({
          title: event == 'add' ? '新增参数信息' : '修改参数信息',
          columns: drawerColumns2.value,
          params: info,
          submit: (value) => {
            if (event == 'add') {
              addEnvironmentItemApi({
                ...value,
                environmentId: currRow.value.id
              }).then(() => {
                msgSuccess('新增成功');
                close();
                createTables2();
              });
            } else {
              updateEnvironmentItemApi(value).then(() => {
                msgSuccess('修改成功');
                close();
                createTables2();
              });
            }
          }
        }
      );
    } else if (event == 'delete') {
      const is = await confirm('确认删除已选中的数据项?', '警告');
      if (is == 'ok') {
        delEnvironmentItemApi(row.id).then(() => {
          msgSuccess('删除成功');
          createTables2();
        });
      }

    }
  }
  ;
  onMounted(() => {
    createTables();
  });
