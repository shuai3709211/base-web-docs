分页页面 ​

搜索+列表

基础用法 ​

  • html部分
  • gs-search标签的search方法后面必须加(true)，否则页码数不会重置
  • gs-table标签的pageChange方法后面必须加()，否则页码会乱
html

<template>
  <div>
    <gs-search v-model="search" v-model:transform-search="transformSearch"
               :columns="searchColumns"
               @search="createTables(true)"></gs-search>
    <gs-table :columns="tableColumns" :data="tables" :loading="loading"
              v-model:page-num="pageNum"
              v-model:page-size="pageSize"
              :total="total"
              @page-change="createTables()"
              curr-table-name="userTable"
              @column-handler="columnHandler" />
  </div>
</template>
  • typescript部分
typescript

import { addApi, delApi, listApi, repeatApi, updateApi } from '@/api/xx/xxxApi';


defineOptions({
  name: 'xxxx'
});


const {
  transformSearch,
  tables,
  pageNum,
  pageSize,
  total,
  loading,
  createTables,
  drawerColumns,
  columnHandler,
  drawerTitle
} = useTablePage({
  apiConfig: {
    loadApi: listApi,
    editApi: updateApi,
    addApi: addApi,
    delApi: delApi

  },

  tableConfig: {
    isPager: true, //分页必须带着
    extendHandler: async ({ event, row, callback }) => {
      if (event == 'add' || event == 'edit') {
        drawerTitle.value = event == 'add' ? '新增' : '修改';
        const response = await listDeptApi({});
        const orgList = response?.data || [];
        drawerColumns.value = [
          {
            el: 'treeSelect',
            prop: 'orgId',
            label: '所属机构',
            checkStrictly: true,
            options: orgList,
            rules: { required: true, message: '请选择所属机构', trigger: 'submit' }
          },
          {
            el: 'select',
            prop: 'gender',
            dictKey: 'dict_common_sex',
            label: '性别',
            rules: { required: true, message: '请选择性别', trigger: 'submit' }
          },
          {
            el: 'input',
            prop: 'account',
            label: '账号',
            rules: {
              required: true, validator: async (_, value, callback) => {
                if (!value) {
                  callback(new Error('请输入账号'));
                } else {
                  const res = await repeatApi(value, row?.userId);
                  if (res.data) {
                    //重复
                    callback(new Error('帐号重复'));
                  }
                }
                callback();
              },
              trigger: 'submit'
            }
          }, {
            el: 'input',
            prop: 'userName',
            label: '姓名',
            rules: { required: true, message: '请输入姓名', trigger: 'submit' }
          },

          {
            el: 'inputNumber',
            prop: 'seq',
            max: 99999,
            min: 0,
            label: '显示排序'
          }
        ];

      }
      await callback(event, row);

      // 除了新增修改删除还有其他的操作就往下写逻辑，如果没有的话删除下面的内容即可
      // 重置密码
      if (event == 'reset_pwd') {
        let is = await confirm(`确定重置用户【${row.userName}】的密码？`);
        if (is == 'ok') {
          resetPwdApi(row.userId).then(({ data }) => {
            confirm(`密码重置成功，新密码是[${data}]，请妥善保管。`, '提示', {
              type: 'success',
              showCancelButton: false
            });
            createTables();
          });
        }
      }

    }
  },
  drawerConfig: {
    // 表单原始内容
    baseRow: {
      gender: 0,
      isDefault: true,
      state: 1
    }
  }
});
