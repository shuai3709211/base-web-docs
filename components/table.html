Table 表格 ​

表格

基础用法 ​

html

<template>
  <gs-table :columns="tableColumns" :data="tables"
            @column-handler="columnHandler" />
</template>

<script setup lang="ts">

  const tableColumns = [
    { prop: 'name', label: '角色名称' },
    { prop: 'code', label: '角色编码' },
    { prop: 'type', label: '角色分类', align: 'center', options: 'dict_sys_role_type' },
    { prop: 'options', label: '操作', width: 240 }
  ];


  const tables = [
    { name: '张三', code: '123', type: '1' },
    { name: '李四', code: '222', type: '3' },
    { name: '王五', code: '333', type: '2' }
  ];
  const columnHandler = (event, row, rowIndex) => {
    console.log('按钮权限值', event);
    console.log('行内容', row);
    console.log('第几行', rowIndex);
  };

</script>
